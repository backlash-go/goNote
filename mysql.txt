批量Kill  查询进程
select concat('kill ', id, ';') from information_schema.processlist where Command != 'Sleep' and Time > 300 order by Time desc;

查询执行的进程按时间排序
select * from information_schema.processlist where Command != 'Sleep' order by Time desc;
